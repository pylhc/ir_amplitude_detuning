1""" 

2Detuning Terms 

3-------------- 

4 

5This module contains the definitions of how the detuning terms are represented 

6in this code and in string outputs (TFS columns, logging, etc.). 

7This follows the PTC-output convention ``ANH[XY] ORDER1 ORDER2 ORDER3 ORDER4``, 

8reduced to the plane of the tune (X or Y) and the order of the first two derivatives ( 

92Jx and 2Jy respectively). 

10 

11This representation is chosen over the LaTeX shorthand "Qxx" to have compact and 

12same-length strings that are still readable. 

13""" 

14from __future__ import annotations 

15 

16from typing import TypeAlias 

17 

18from ir_amplitude_detuning.utilities.common import StrEnum 

19 

20 

21class FirstOrderTerm(StrEnum): 

22 X10: str = "X10" # d Qx / d 2Jx 

23 X01: str = "X01" # d Qx / d 2Jy 

24 Y10: str = "Y10" # d Qy / d 2Jx 

25 Y01: str = "Y01" # d Qy / d 2Jy 

26 

27 

28class SecondOrderTerm(StrEnum): 

29 X20: str = "X20" # d^2 Qx / (d 2Jx)^2 

30 X11: str = "X11" # d^2 Qx / (d 2Jx)(d Jy) 

31 X02: str = "X02" # d^2 Qx / (d 2Jy)^2 

32 Y20: str = "Y20" # d^2 Qy / (d 2Jx)^2 

33 Y11: str = "Y11" # d^2 Qy / (d 2Jx)(d Jy) 

34 Y02: str = "Y02" # d^2 Qy / (d 2Jy)^2 

35 

36 

37DetuningTerm: TypeAlias = FirstOrderTerm | SecondOrderTerm 

38 

39 

40def get_order(term: DetuningTerm) -> int: 

41 """Get the order of the detuning, e.g. from X11 -> order 2, Y10 -> order 1. 

42 

43 Args: 

44 term (str): Detuning Term, e.g. "X11" 

45 """ 

46 return int(term[1]) + int(term[2]) 

47 

48 

49def detuning_term_to_planes(term: DetuningTerm) -> tuple[str, str]: 

50 """Get the tune and action planes given detuning term. 

51 

52 Args: 

53 term (str): Detuning term, e.g. "X02" 

54 

55 Returns: 

56 tuple[str, str]: (tune, action), e.g. ("x", "yy") 

57 """ 

58 tune = term[0].lower() 

59 action = "x" * int(term[1]) + "y" * int(term[2]) 

60 return tune, action
