1""" 

2LHC Simulation Class 

3-------------------- 

4 

5Run a cpymad MAD-X simulation for the LHC optics (2018) without errors. 

6In addition, extra functionality is added to install kcdx decapole correctors 

7into the MCTX and assign powering for decapole and dodecapole circuits. 

8 

9The class ``LHCBeam`` is setting up and running cpymad. 

10This class can be useful for a lot of different studies, by extending 

11it with extra functionality.

61def drop_allzero_columns(df: TfsDataFrame, keep: Sequence = ()) -> TfsDataFrame: 

62 """Drop columns that contain only zeros, to save harddrive space. 

63 

64 Args:

74class LHCCorrectors: 

75 """Container for the corrector definitions used in the LHC. 

76 

77 As the LHC does not have decapoe correctors, all correctors are 

78 installed into the MCTX and powered via kcdx3 and kctx3 circuits. 

79 The decapole correctors are hence only used for simulation purposes. 

80 For HiLumi, which has more actual IR correctors that can be used, 

81 you will need to adapt the simulation and correctors a bit, 

82 but it should be straightforward. 

83 

84 Note that in the correction algorithm only the normal-oriented 

85 fields are implemented. You will need to add a5 if you are planning 

86 on using this corrector. 

87 

88 The length is set to 0.615 m, which is the length of the MCTs. 

89 The pattern is used to find the correctors in the MAD-X sequence. 

90 """ 

91 b5 = CorrectorMask( 

92 field=FieldComponent.b5, 

93 length=0.615, 

94 magnet_pattern="MCTX.3{side}{ip}", # installed into the MCTX 

95 circuit_pattern="kcdx3.{side}{ip}", 

96 madx_type="MCTX", 

97 ) 

98 b6 = CorrectorMask( 

99 field=FieldComponent.b6, 

100 length=0.615, 

101 magnet_pattern="MCTX.3{side}{ip}", 

102 circuit_pattern="kctx3.{side}{ip}", 

103 madx_type="MCTX", 

104 ) 

105 pattern: str = "MCTX.*[15]$" # used to find correctors in twiss-table

133 def __post_init__(self): 

134 """Setup the MADX, output dirs and logging as well as additional instance parameters.""" 

135 self.outputdir.mkdir(exist_ok=True, parents=True) 

136 self.madx = Madx( 

137 **cpymad_logging_setup( # sets also standard loggers 

138 level=LOG_LEVEL, 

139 command_log=self.outputdir / "madx_commands.log", 

140 full_log=self.outputdir / "full_output.log", 

141 ), 

142 cwd=self.outputdir, 

143 ) 

144 self.logger = {key: logging.getLogger(key).handlers for key in ("", MADXOUT, MADXCMD)} # save logger to reinstate later 

145 

146 if self.beam == 2: 

147 LOG.debug("Input as Beam 2 detected. Running with Beam 4.") 

148 self.beam = 4 

149 

150 self.madx.globals.mylhcbeam = self.beam # used in macros 

151 

152 # Define Sequence to use 

153 self.seq_name, self.seq_file, self.bv_flag = get_lhc_sequence_filename_and_bv(self.beam, accel="lhc" if self.year < 2020 else "hllhc") # `hllhc` just for naming of the sequence file, i.e. without _as_built 

154 

155 # Setup Model Paths (always use acc-models-like symlink) 

156 if self.year < 2015: 

157 raise NotImplementedError("LHC models before run II are not implemented in this simulation.") 

158 

159 model_src = PATHS["acc_models_lhc" if self.year > 2019 else "optics_runII"] / str(self.year) 

160 acc_models_path = self.outputdir / ACC_MODELS 

161 if acc_models_path.exists(): 

162 acc_models_path.unlink() 

163 acc_models_path.symlink_to(model_src)

166 def output_path(self, type_: str, output_id: str, dir_: Path | None = None, suffix: str = ".tfs") -> Path: 

167 """Returns the output path for standardized tfs names in the default output directory. 

168 

169 Args:

171 output_id (str): Name of the output (e.g. 'nominal') 

172 dir_ (Path): Override default directory. 

173 suffix (str): suffix of the output file. 

174 

175 Returns: 

176 Path: Path to the output file 

177 """ 

178 if dir_ is None: 

179 dir_ = self.outputdir 

180 return dir_ / f'{type_}.lhc.b{self.beam:d}.{output_id}{suffix}'

182 def get_twiss(self, output_id=None, index_regex=r"BPM|M|IP", **kwargs) -> TfsDataFrame: 

183 """Uses the ``twiss`` command to get the current optics in the machine 

184 as TfsDataFrame. 

185 

186 Args: 

187 output_id (str): ID to use in the output (see ``output_path``). 

188 If not given, no output is written. 

189 index_regex (str): Filter DataFrame index (NAME) by this pattern. 

190 

191 Returns: 

192 TfsDataFrame: DataFrame containing the optics. 

193 """ 

194 kwargs['chrom'] = kwargs.get('chrom', True) 

195 kwargs['centre'] = kwargs.get('centre', True) 

196 self.madx.twiss(sequence=self.seq_name, **kwargs) 

197 df_twiss = self.get_last_twiss(index_regex=index_regex) 

198 if output_id is not None: 

199 self.write_tfs(df_twiss, 'twiss', output_id) 

200 return df_twiss

287 def setup_machine(self): 

288 """Nominal machine setup function. 

289 Initialized the beam and applies optics, crossing.""" 

290 self.reinstate_loggers() 

291 madx = self.madx # shorthand 

292 

293 # suppress output from reading files, which leads to much smaller test-logs 

294 # e.g. in the CI and allows for actual test-debugging. 

295 # Note: I am locally running into mad-x memory-out-of-scope errors in the madx.seqedit 

296 # below with the two madx.options commands here. No idea why. It works in the CI. 

297 madx.option(echo=False) 

298 

299 # Load Macros 

300 madx.call(pathstr("toolkit", "macro.madx")) 

301 

302 # Lattice Setup --------------------------------------- 

303 # Load Sequence 

304 madx.call(pathstr(self.seq_file)) 

305 

306 # re-enable output 

307 madx.option(echo=True) 

308 

309 # Cycling w.r.t. to IP3 (mandatory to find closed orbit in collision in the presence of errors) 

310 madx.seqedit(sequence=self.seq_name) 

311 madx.flatten() 

312 madx.cycle(start="IP3") 

313 madx.endedit() 

314 

315 # Define Optics and make beam 

316 madx.call(str(self.optics)) 

317 madx.beam(sequence=self.seq_name, bv=self.bv_flag, 

318 energy="NRJ", particle="proton", npart=self.n_particles, 

319 kbunch=1, ex=self.emittance, ey=self.emittance) 

320 

321 # Setup Orbit 

322 orbit_setup(madx, accel='lhc', year=self.year, **self.xing) 

323 

324 madx.use(sequence=self.seq_name)

342 def install_circuits_into_mctx(self): 

343 """Installs kcdx and (and reinstalls kctx) into the Dodecapole Correctors. 

344 

345 This allows for decapole and dodecapole correction with the MCTX magnets for test purposes. 

346 """ 

347 self.reinstate_loggers() 

348 beam_sign_str = "-" if self.beam == 4 else "" 

349 for ip in (1, 5): 

350 for side in "LR": 

351 corrector_b5 = LHCCorrectors.b5.get_corrector(side, ip) 

352 corrector_b6 = LHCCorrectors.b6.get_corrector(side, ip) 

353 

354 assert corrector_b5.magnet == corrector_b6.magnet, "Magnet name for b5 and b6 must be the same as we install b5 corrector in same magnet!" 

355 

356 deca_knl = f"{corrector_b5.circuit} * l.{corrector_b5.madx_type}" 

357 dodeca_knl = f"{beam_sign_str}{corrector_b6.circuit} * l.{corrector_b6.madx_type}" 

358 

359 self.madx.input(f"{corrector_b6.magnet}, KNL := {{0, 0, 0, 0, {deca_knl}, {dodeca_knl}}}, polarity=+1;") 

360 self.madx.globals[corrector_b5.circuit] = 0 

361 self.madx.globals[corrector_b6.circuit] = 0

363 def reset_detuning_circuits(self): 

364 """Reset all kcdx and kctx circuits (to zero).""" 

365 for ip in (1, 5): 

366 for side in "LR": 

367 for corrector_mask in (LHCCorrectors.b5, LHCCorrectors.b6): 

368 corrector = corrector_mask.get_corrector(side, ip) 

369 self.madx.globals[corrector.circuit] = 0
